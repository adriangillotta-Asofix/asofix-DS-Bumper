Checkbox & Radio

Preview

Checkbox

Permite seleccionar una o varias opciones dentro de un grupo.

Unchecked

Intermediate

Checked

.checkbox {
  position: relative;
  display: inline-flex;
  align-items: center;
  gap: 4px;
  cursor: pointer;
}

.checkbox__input {
  position: absolute;
  opacity: 0;
  width: 32px;
  height: 32px;
  margin: 0;
  cursor: pointer;
}

.checkbox__icon {
  display: inline-flex;
  align-items: center;
  justify-content: center;
  width: 32px;
  height: 32px;
  border-radius: var(--radius-full);
  font-size: 20px;
  color: var(--color-neutral-300);
  transition: background-color 0.15s ease, color 0.15s ease;
}

.checkbox__label {
  font-family: var(--font-family-base);
  font-size: var(--font-size-body);
  color: var(--color-neutral-600);
}

.checkbox:hover .checkbox__icon { background-color: var(--color-primary-50); }

.checkbox__input:checked ~ .checkbox__icon,
.checkbox--indeterminate .checkbox__icon { color: var(--color-primary-500); }

.checkbox__input:focus-visible ~ .checkbox__icon {
  outline: 2px solid var(--color-primary-300);
  outline-offset: 2px;
}

.checkbox__input:disabled ~ .checkbox__icon { color: var(--color-neutral-200); }
.checkbox__input:disabled:checked ~ .checkbox__icon,
.checkbox--indeterminate.checkbox--disabled .checkbox__icon { color: var(--color-neutral-400); }
<!-- Unchecked / Checked: el input real controla el estado -->
<label class="checkbox">
  <input class="checkbox__input" type="checkbox" onchange="syncCheckboxIcon(this)" />
  <span class="checkbox__icon material-icons-outlined">check_box_outline_blank</span>
  <span class="checkbox__label">Checkbox text</span>
</label>

<!-- Intermediate: la propiedad .indeterminate no es un atributo HTML, se setea por JS -->
<label class="checkbox checkbox--indeterminate">
  <input class="checkbox__input" type="checkbox" onchange="setCheckboxIndeterminate(this, false)" />
  <span class="checkbox__icon material-icons-outlined">indeterminate_check_box</span>
</label>
<script>setCheckboxIndeterminate(input, true)</script>
