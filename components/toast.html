const DURATION = 4000;

function showToast(type) {
  const toast = document.createElement('div');
  toast.className = `toast toast--${type} toast--in`;
  toast.innerHTML = `...`; // ver tab HTML

  // Timer con pausa en hover
  let remaining = DURATION;
  let startTime = Date.now();
  let timerId   = setTimeout(() => dismissToast(toast), DURATION);

  toast.addEventListener('mouseenter', () => {
    clearTimeout(timerId);
    remaining -= Date.now() - startTime;
  });

  toast.addEventListener('mouseleave', () => {
    startTime = Date.now();
    timerId   = setTimeout(() => dismissToast(toast), remaining);
  });

  const stack = document.getElementById('toast-stack');
  stack.insertBefore(toast, stack.firstChild);
}

function dismissToast(toast) {
  if (!toast || toast.classList.contains('toast--out')) return;
  toast.classList.remove('toast--in');
  toast.classList.add('toast--out');
  setTimeout(() => toast.remove(), 250);
}
