function openModal() {
  const overlay = document.getElementById('modal-overlay');
  overlay.hidden = false;
  overlay.classList.remove('modal-overlay--out');
  overlay.querySelector('.modal').classList.remove('modal--out');
  overlay.classList.add('modal-overlay--in');
  overlay.querySelector('.modal').classList.add('modal--in');
  document.addEventListener('keydown', handleModalEsc);
}

function closeModal() {
  const overlay = document.getElementById('modal-overlay');
  overlay.classList.replace('modal-overlay--in', 'modal-overlay--out');
  overlay.querySelector('.modal').classList.replace('modal--in', 'modal--out');
  setTimeout(() => {
    overlay.hidden = true;
    overlay.classList.remove('modal-overlay--out');
    overlay.querySelector('.modal').classList.remove('modal--out');
  }, 200);
  document.removeEventListener('keydown', handleModalEsc);
}

function handleModalEsc(e) {
  if (e.key === 'Escape') closeModal();
}
